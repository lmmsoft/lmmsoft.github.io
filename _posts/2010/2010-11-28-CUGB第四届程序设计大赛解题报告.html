D题：（哈尔滨现场赛的题目）贪心算法，排序，正逆相加即可

#include<iostream>
#include<algorithm>
using namespace std;
int a[1010],b[1010];
int main(){
int t,tt,n,x;
int s,ans;
while(scanf("%d%d",&n,&x)!=EOF)
{
   for(tt=0;tt<n;tt++)
   {
    scanf("%d",&a[tt]);

   }
   for(tt=0;tt<n;tt++)
   {
    scanf("%d",&b[tt]);

   }
   sort(a,a+n);
   sort(b,b+n);
   ans=0;
   for(tt=0;tt<n;tt++)
   {
    s=a[tt]+b[n-tt-1]-x;
    if(s>0)
     ans+=s;
   }
   printf("%d\n",ans);
}
    return 0;
}

+2*9+7 13-16张牌：4n+3*12+2*9+7 17-21张牌：5n+4*16+3*12+2*9+7 可以发

现区间段的数字个数为 2,3,4,5,6....（从数字8开始）;可以发现得分的常数

项为7,2*9+7,3*12+2*9+7,....(从数字8开始）：每一个常数项和上一个数段
